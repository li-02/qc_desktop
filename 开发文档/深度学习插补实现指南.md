## 1. 概述

### 1.1 目标

本文档指导你实现**深度学习模型**（LSTM、GRU、Transformer 等）用于缺失值数据插补。主要工作包括：

1. 在系统中集成 Python 深度学习环境
2. 开发深度学习插补脚本
3. 将深度学习方法集成到现有的插补流程中

### 1.2 前置条件

- 熟悉当前项目的 Electron + Vue 架构
- 已了解现有的缺失值插补模块（参见 `缺失值插补方案设计.md`）
- 基本的 Python 和 PyTorch/TensorFlow 知识

### 1.3 涉及的深度学习方法

| 方法 ID | 方法名称 | 描述 | 适用场景 |
|---------|----------|------|----------|
| `LSTM` | LSTM 网络 | 长短期记忆网络，处理时序依赖 | 时间序列数据 |
| `GRU` | GRU 网络 | 门控循环单元，LSTM 的简化版 | 时间序列数据 |
| `TRANSFORMER` | Transformer | 基于注意力机制的模型 | 长序列时序数据 |
| `VAE` | 变分自编码器 | 生成模型，学习数据分布 | 高维数据 |
| `GAIN` | GAIN 网络 | 生成对抗插补网络 | 复杂缺失模式 |

```python
#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""
深度学习缺失值插补脚本

支持的模型:
- LSTM: 长短期记忆网络
- GRU: 门控循环单元
- TRANSFORMER: Transformer 模型

使用方式:
python dl_imputation.py --file data.csv --column Temperature --method LSTM [其他参数]
"""

import argparse
import json
import sys
import os
import traceback
from typing import Dict, Any, List, Optional, Tuple
from datetime import datetime

import numpy as np
import pandas as pd
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, Dataset
from sklearn.preprocessing import StandardScaler


# ==================== 数据处理类 ====================

class TimeSeriesDataset(Dataset):
    """时间序列数据集"""
    
    def __init__(self, data: np.ndarray, window_size: int = 24):
        self.data = data
        self.window_size = window_size
        self.valid_indices = self._get_valid_indices()
        
    def _get_valid_indices(self) -> List[int]:
        """获取有效的训练索引（前后都有足够数据点）"""
        indices = []
        for i in range(self.window_size, len(self.data) - 1):
            # 检查窗口内是否有足够的非缺失值
            window = self.data[i - self.window_size:i]
            if not np.isnan(window).all():
                indices.append(i)
        return indices
    
    def __len__(self):
        return len(self.valid_indices)
    
    def __getitem__(self, idx: int) -> Tuple[torch.Tensor, torch.Tensor]:
        real_idx = self.valid_indices[idx]
        window = self.data[real_idx - self.window_size:real_idx].copy()
        target = self.data[real_idx]
        
        # 用均值填充窗口内的缺失值
        mask = np.isnan(window)
        if mask.any():
            window[mask] = np.nanmean(window)
            if np.isnan(window).all():
                window[:] = 0
        
        return (
            torch.FloatTensor(window).unsqueeze(-1),  # (window_size, 1)
            torch.FloatTensor([target if not np.isnan(target) else 0])
        )


# ==================== 模型定义 ====================

class LSTMModel(nn.Module):
    """LSTM 插补模型"""
    
    def __init__(
        self, 
        input_size: int = 1,
        hidden_size: int = 64,
        num_layers: int = 2,
        dropout: float = 0.2
    ):
        super().__init__()
        self.lstm = nn.LSTM(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0
        )
        self.fc = nn.Linear(hidden_size, 1)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: (batch, window_size, input_size)
        lstm_out, _ = self.lstm(x)
        # 取最后一个时间步
        out = self.fc(lstm_out[:, -1, :])
        return out


class GRUModel(nn.Module):
    """GRU 插补模型"""
    
    def __init__(
        self,
        input_size: int = 1,
        hidden_size: int = 64,
        num_layers: int = 2,
        dropout: float = 0.2
    ):
        super().__init__()
        self.gru = nn.GRU(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0
        )
        self.fc = nn.Linear(hidden_size, 1)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        gru_out, _ = self.gru(x)
        out = self.fc(gru_out[:, -1, :])
        return out


class TransformerModel(nn.Module):
    """Transformer 插补模型"""
    
    def __init__(
        self,
        input_size: int = 1,
        d_model: int = 64,
        nhead: int = 4,
        num_layers: int = 2,
        dropout: float = 0.1
    ):
        super().__init__()
        self.embedding = nn.Linear(input_size, d_model)
        self.pos_encoder = self._generate_positional_encoding(d_model, 1000)
        
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model,
            nhead=nhead,
            dropout=dropout,
            batch_first=True
        )
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers=num_layers)
        self.fc = nn.Linear(d_model, 1)
    
    def _generate_positional_encoding(self, d_model: int, max_len: int) -> torch.Tensor:
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * (-np.log(10000.0) / d_model))
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        return pe.unsqueeze(0)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: (batch, window_size, input_size)
        seq_len = x.size(1)
        x = self.embedding(x)  # (batch, window_size, d_model)
        x = x + self.pos_encoder[:, :seq_len, :].to(x.device)
        x = self.transformer(x)
        out = self.fc(x[:, -1, :])
        return out


# ==================== 训练器类 ====================

class DLImputer:
    """深度学习插补器"""
    
    MODEL_CLASSES = {
        'LSTM': LSTMModel,
        'GRU': GRUModel,
        'TRANSFORMER': TransformerModel
    }
    
    def __init__(
        self,
        method: str = 'LSTM',
        window_size: int = 24,
        hidden_size: int = 64,
        num_layers: int = 2,
        epochs: int = 100,
        learning_rate: float = 0.001,
        batch_size: int = 32,
        early_stopping_patience: int = 10,
        device: str = 'auto'
    ):
        self.method = method.upper()
        self.window_size = window_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.epochs = epochs
        self.learning_rate = learning_rate
        self.batch_size = batch_size
        self.early_stopping_patience = early_stopping_patience
        
        # 设置设备
        if device == 'auto':
            self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        else:
            self.device = torch.device(device)
        
        self.model = None
        self.scaler = StandardScaler()
        self.training_history = []
    
    def _build_model(self) -> nn.Module:
        """构建模型"""
        model_class = self.MODEL_CLASSES.get(self.method)
        if model_class is None:
            raise ValueError(f"不支持的方法: {self.method}")
        
        if self.method == 'TRANSFORMER':
            return model_class(
                input_size=1,
                d_model=self.hidden_size,
                nhead=4,
                num_layers=self.num_layers
            ).to(self.device)
        else:
            return model_class(
                input_size=1,
                hidden_size=self.hidden_size,
                num_layers=self.num_layers
            ).to(self.device)
    
    def fit(self, data: np.ndarray, progress_callback=None) -> Dict[str, Any]:
        """训练模型"""
        # 数据标准化
        data_scaled = self.scaler.fit_transform(data.reshape(-1, 1)).flatten()
        
        # 创建数据集
        dataset = TimeSeriesDataset(data_scaled, self.window_size)
        if len(dataset) < 10:
            raise ValueError("有效训练样本不足，请检查数据或减小窗口大小")
        
        dataloader = DataLoader(dataset, batch_size=self.batch_size, shuffle=True)
        
        # 构建模型
        self.model = self._build_model()
        optimizer = torch.optim.Adam(self.model.parameters(), lr=self.learning_rate)
        criterion = nn.MSELoss()
        
        # 训练循环
        best_loss = float('inf')
        patience_counter = 0
        
        for epoch in range(self.epochs):
            self.model.train()
            epoch_loss = 0.0
            batch_count = 0
            
            for batch_x, batch_y in dataloader:
                batch_x = batch_x.to(self.device)
                batch_y = batch_y.to(self.device)
                
                optimizer.zero_grad()
                outputs = self.model(batch_x)
                loss = criterion(outputs, batch_y)
                loss.backward()
                optimizer.step()
                
                epoch_loss += loss.item()
                batch_count += 1
            
            avg_loss = epoch_loss / batch_count
            self.training_history.append({'epoch': epoch + 1, 'loss': avg_loss})
            
            # 早停检查
            if avg_loss < best_loss:
                best_loss = avg_loss
                patience_counter = 0
            else:
                patience_counter += 1
                if patience_counter >= self.early_stopping_patience:
                    if progress_callback:
                        progress_callback(f"早停于第 {epoch + 1} 轮")
                    break
            
            # 进度回调
            if progress_callback and epoch % 10 == 0:
                progress_callback(f"训练进度: {epoch + 1}/{self.epochs}, 损失: {avg_loss:.6f}")
        
        return {
            'final_loss': best_loss,
            'epochs_trained': len(self.training_history),
            'training_history': self.training_history
        }
    
    def predict(self, data: np.ndarray, missing_indices: List[int]) -> np.ndarray:
        """预测缺失值"""
        if self.model is None:
            raise ValueError("模型尚未训练")
        
        self.model.eval()
        data_scaled = self.scaler.transform(data.reshape(-1, 1)).flatten()
        result = data.copy()
        
        with torch.no_grad():
            for idx in sorted(missing_indices):
                if idx >= self.window_size:
                    # 获取窗口数据
                    window = data_scaled[idx - self.window_size:idx].copy()
                    
                    # 填充窗口内的缺失值
                    mask = np.isnan(window)
                    if mask.any():
                        window[mask] = np.nanmean(window)
                        if np.isnan(window).all():
                            window[:] = 0
                    
                    # 预测
                    x = torch.FloatTensor(window).unsqueeze(0).unsqueeze(-1).to(self.device)
                    pred = self.model(x).cpu().numpy()[0, 0]
                    
                    # 反标准化
                    pred_original = self.scaler.inverse_transform([[pred]])[0, 0]
                    result[idx] = pred_original
                    data_scaled[idx] = pred  # 更新用于后续预测
                    
                else:
                    # 窗口不足，使用均值填充
                    valid_values = data[~np.isnan(data)]
                    if len(valid_values) > 0:
                        result[idx] = np.mean(valid_values)
                    else:
                        result[idx] = 0
        
        return result
    
    def save_model(self, path: str):
        """保存模型"""
        torch.save({
            'model_state_dict': self.model.state_dict(),
            'scaler_mean': self.scaler.mean_,
            'scaler_scale': self.scaler.scale_,
            'config': {
                'method': self.method,
                'window_size': self.window_size,
                'hidden_size': self.hidden_size,
                'num_layers': self.num_layers
            }
        }, path)
    
    def load_model(self, path: str):
        """加载模型"""
        checkpoint = torch.load(path, map_location=self.device)
        self.method = checkpoint['config']['method']
        self.window_size = checkpoint['config']['window_size']
        self.hidden_size = checkpoint['config']['hidden_size']
        self.num_layers = checkpoint['config']['num_layers']
        
        self.model = self._build_model()
        self.model.load_state_dict(checkpoint['model_state_dict'])
        
        self.scaler.mean_ = checkpoint['scaler_mean']
        self.scaler.scale_ = checkpoint['scaler_scale']


# ==================== 主函数 ====================

def parse_missing_value(value: Any, missing_types: List[str]) -> bool:
    """判断值是否为缺失值"""
    if pd.isna(value):
        return True
    if isinstance(value, str):
        return value.lower() in [m.lower() for m in missing_types]
    return False

    parser.add_argument('--early-stopping', type=int, default=10, help='早停耐心值')
    
    # 输出选项
    parser.add_argument('--output-file', help='输出文件路径（默认为原文件名_imputed）')
    parser.add_argument('--save-model', help='保存模型路径')
    parser.add_argument('--load-model', help='加载已有模型路径')
    
    args = parser.parse_args()
    
    result = {
        'success': False,
        'message': '',
        'imputed_count': 0,
        'output_file': '',
        'statistics': {},
        'model_info': {}
    }
    
    try:
        # 1. 读取数据
        if args.file.endswith('.xlsx') or args.file.endswith('.xls'):
            df = pd.read_excel(args.file)
        else:
            df = pd.read_csv(args.file)
        
        if args.column not in df.columns:
            raise ValueError(f"列 '{args.column}' 不存在于数据中")
        
        # 2. 识别缺失值
        original_column = df[args.column].copy()
        missing_mask = df[args.column].apply(lambda x: parse_missing_value(x, args.missing_types))
        missing_indices = df.index[missing_mask].tolist()
        
        if len(missing_indices) == 0:
            result['success'] = True
            result['message'] = '该列没有缺失值'
            result['imputed_count'] = 0
            print(json.dumps(result, ensure_ascii=False))
            return
        
        # 转换为数值类型
        data = pd.to_numeric(df[args.column], errors='coerce').values
        
        valid_count = np.sum(~np.isnan(data))
        if valid_count < 20:
            raise ValueError(f"有效数据点太少 ({valid_count})，需要至少20个数据点")
        
        # 3. 创建插补器
        imputer = DLImputer(
            method=args.method,
            window_size=args.window_size,
            hidden_size=args.hidden_size,
            num_layers=args.num_layers,
            epochs=args.epochs,
            learning_rate=args.learning_rate,
            batch_size=args.batch_size,
            early_stopping_patience=args.early_stopping
        )
        
        # 4. 训练或加载模型
        if args.load_model and os.path.exists(args.load_model):
            imputer.load_model(args.load_model)
            training_info = {'loaded_from': args.load_model}
        else:
            def progress_callback(msg):
                print(f"[PROGRESS] {msg}", file=sys.stderr)
            
            training_info = imputer.fit(data, progress_callback)
        
        # 5. 预测缺失值
        imputed_data = imputer.predict(data, missing_indices)
        
        # 6. 更新数据
        df[args.column] = imputed_data
        
        # 7. 计算统计信息
        valid_original = original_column[~missing_mask].astype(float)
        imputed_values = imputed_data[missing_indices]
        
        statistics = {
            'mean_before': float(valid_original.mean()) if len(valid_original) > 0 else None,
            'std_before': float(valid_original.std()) if len(valid_original) > 0 else None,
            'mean_after': float(np.mean(imputed_data)),
            'std_after': float(np.std(imputed_data)),
            'imputed_min': float(np.min(imputed_values)),
            'imputed_max': float(np.max(imputed_values)),
            'imputed_mean': float(np.mean(imputed_values))
        }
        
        # 8. 保存结果
        output_file = args.output_file
        if not output_file:
            base, ext = os.path.splitext(args.file)
            output_file = f"{base}_imputed{ext}"
        
        if output_file.endswith('.xlsx') or output_file.endswith('.xls'):
            df.to_excel(output_file, index=False)
        else:
            df.to_csv(output_file, index=False)
        
        # 9. 保存模型（可选）
        if args.save_model:
            imputer.save_model(args.save_model)
        
        # 10. 返回结果
        result['success'] = True
        result['message'] = f'成功使用 {args.method} 插补 {len(missing_indices)} 个缺失值'
        result['imputed_count'] = len(missing_indices)
        result['missing_rate'] = len(missing_indices) / len(data) * 100
        result['output_file'] = os.path.abspath(output_file)
        result['statistics'] = statistics
        result['model_info'] = {
            'method': args.method,
            'window_size': args.window_size,
            'hidden_size': args.hidden_size,
            'num_layers': args.num_layers,
            'device': str(imputer.device),
            'training_epochs': training_info.get('epochs_trained', 0),
            'final_loss': training_info.get('final_loss', None)
        }
        
    except Exception as e:
        result['success'] = False
        result['message'] = str(e)
        result['error_traceback'] = traceback.format_exc()
    
    print(json.dumps(result, ensure_ascii=False, indent=2))


if __name__ == '__main__':
    main()
```

```typescript
/**
 * 执行深度学习插补
 */
private async performDLImputation(
  resultId: number,
  data: Record<string, any>[],
  targetColumns: string[],
  methodId: ImputationMethodId,
  params: Record<string, any>,
  filePath: string,
  missingMarkers: string[]
): Promise<{
  imputedData: Record<string, any>[];
  columnStats: Map<string, {
    missingCount: number;
    imputedCount: number;
    imputationRate: number;
    meanBefore?: number;
    meanAfter?: number;
    stdBefore?: number;
    stdAfter?: number;
    minImputed?: number;
    maxImputed?: number;
  }>;
}> {
  const columnStats = new Map<string, any>();
  const imputedData = [...data];
  
  // 发送准备阶段进度
  this.emitProgress(resultId, {
    resultId,
    stage: 'preparing',
    progress: 5,
    message: '正在准备深度学习环境...'
  });
  
  // 逐列处理
  const totalColumns = targetColumns.length;
  
  for (let colIdx = 0; colIdx < totalColumns; colIdx++) {
    const columnName = targetColumns[colIdx];
    const baseProgress = 10 + (colIdx / totalColumns) * 80;
    
    // 发送训练阶段进度
    this.emitProgress(resultId, {
      resultId,
      stage: 'training',
      progress: Math.round(baseProgress),
      message: `正在训练 ${methodId} 模型...`,
      currentColumn: columnName,
      processedColumns: colIdx,
      totalColumns
    });
    
    try {
      // 调用 Python 脚本
      const result = await this.callDLPythonScript(
        filePath,
        columnName,
        methodId,
        params,
        missingMarkers
      );
      
      if (!result.success) {
        throw new Error(result.message || '深度学习插补失败');
      }
      
      // 读取插补后的数据
      const imputedColumnData = await this.readCsvFile(result.output_file);
      
      // 更新数据
      for (let i = 0; i < imputedColumnData.length && i < imputedData.length; i++) {
        imputedData[i][columnName] = imputedColumnData[i][columnName];
      }
      
      // 记录统计信息
      columnStats.set(columnName, {
        missingCount: result.imputed_count,
        imputedCount: result.imputed_count,
        imputationRate: result.imputed_count > 0 ? 1.0 : 0,
        meanBefore: result.statistics?.mean_before,
        meanAfter: result.statistics?.mean_after,
        stdBefore: result.statistics?.std_before,
        stdAfter: result.statistics?.std_after,
        minImputed: result.statistics?.imputed_min,
        maxImputed: result.statistics?.imputed_max
      });
      
    } catch (error: any) {
      console.error(`列 ${columnName} 深度学习插补失败:`, error);
      // 回退到线性插值
      columnStats.set(columnName, {
        missingCount: 0,
        imputedCount: 0,
        imputationRate: 0
      });
    }
    
    // 发送插补阶段进度
    this.emitProgress(resultId, {
      resultId,
      stage: 'imputing',
      progress: Math.round(baseProgress + 10),
      message: `已完成 ${columnName} 的插补`,
      currentColumn: columnName,
      processedColumns: colIdx + 1,
      totalColumns
    });
  }
  
  return { imputedData, columnStats };
}
```

#### 6.3.3 添加 Python 脚本调用方法

```typescript
import { spawn } from 'child_process';
import * as path from 'path';

/**
 * 调用深度学习 Python 脚本
 */
private callDLPythonScript(
  filePath: string,
  columnName: string,
  methodId: string,
  params: Record<string, any>,
  missingMarkers: string[]
): Promise<{
  success: boolean;
  message: string;
  imputed_count: number;
  output_file: string;
  statistics: any;
  model_info: any;
}> {
  return new Promise((resolve, reject) => {
    // Python 脚本路径
    const scriptPath = path.join(__dirname, '../../python/dl_imputation.py');
    
    // 构建命令行参数
    const args = [
      scriptPath,
      '--file', filePath,
      '--column', columnName,
      '--method', methodId
    ];
    
    // 添加缺失值标记
    if (missingMarkers.length > 0) {
      args.push('--missing-types', ...missingMarkers);
    }
    
    // 添加模型参数
    if (params.window_size) {
      args.push('--window-size', String(params.window_size));
    }
    if (params.hidden_size) {
      args.push('--hidden-size', String(params.hidden_size));
    }
    if (params.num_layers) {
      args.push('--num-layers', String(params.num_layers));
    }
    if (params.epochs) {
      args.push('--epochs', String(params.epochs));
    }
    if (params.learning_rate) {
      args.push('--learning-rate', String(params.learning_rate));
    }
    
    // 输出文件路径
    const outputDir = path.dirname(filePath);
    const baseName = path.basename(filePath, path.extname(filePath));
    const outputFile = path.join(outputDir, `${baseName}_${columnName}_imputed.csv`);
    args.push('--output-file', outputFile);
    
    // Python 项目目录（用于 uv run 的工作目录）
    const pythonDir = path.join(__dirname, '../../python');
    
    console.log('执行 Python 脚本 (via uv):', 'uv run python', args.join(' '));
    
    // 使用 uv run 启动 Python 进程
    const python = spawn('uv', ['run', 'python', ...args], {
      cwd: pythonDir
    });
    
    let stdout = '';
    let stderr = '';
    
    python.stdout.on('data', (data) => {
      stdout += data.toString();
    });
    
    python.stderr.on('data', (data) => {
      const msg = data.toString();
      stderr += msg;
      // 检查是否是进度信息
      if (msg.includes('[PROGRESS]')) {
        console.log('Python 进度:', msg);
      }
    });
    
    python.on('close', (code) => {
      if (code !== 0) {
        console.error('Python 脚本错误:', stderr);
        reject(new Error(`Python 脚本退出码: ${code}, 错误: ${stderr}`));
        return;
      }
      
      try {
        const result = JSON.parse(stdout);
        resolve(result);
      } catch (e) {
        reject(new Error(`解析 Python 输出失败: ${stdout}`));
      }
    });
    
    python.on('error', (error) => {
      reject(error);
    });
  });
}
```
