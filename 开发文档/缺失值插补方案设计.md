# 缺失值插补方案设计文档

## 1. 概述

### 1.1 目标
设计一个功能完整、用户友好的缺失值插补模块，支持多种插补方法（普通统计方法、机器学习模型、深度学习模型），提供直观的可视化对比和实时进度反馈。

### 1.2 核心功能
- **插补记录管理**：展示历史插补记录，支持查看、删除、回滚
- **多方法支持**：普通方法、机器学习模型、深度学习模型
- **智能模型匹配**：根据数据集类型自动推荐适配模型
- **灵活列选择**：全列插补或手动选择指定列
- **可视化对比**：插补前后数据对照图表
- **进度反馈**：实时展示插补过程和进度

## 3. 插补方法定义

### 3.1 方法分类

```typescript
export type ImputationMethodId = 
  // 基础方法
  | 'MEAN'              // 均值插补
  | 'MEDIAN'            // 中位数插补
  | 'MODE'              // 众数插补
  | 'FORWARD_FILL'      // 前向填充
  | 'BACKWARD_FILL'     // 后向填充
  // 统计/插值方法
  | 'LINEAR'            // 线性插值
  | 'SPLINE'            // 样条插值
  | 'POLYNOMIAL'        // 多项式插值
  | 'SEASONAL'          // 季节性分解插补
  // 时间序列模型
  | 'ARIMA'             // ARIMA模型
  | 'SARIMA'            // 季节性ARIMA
  | 'ETS'               // 指数平滑
  // 机器学习模型
  | 'KNN'               // K近邻
  | 'RANDOM_FOREST'     // 随机森林
  | 'GRADIENT_BOOSTING' // 梯度提升
  | 'MICE'              // 多重插补
  | 'MISSFOREST'        // MissForest
  // 深度学习模型
  | 'LSTM'              // LSTM网络
  | 'GRU'               // GRU网络
  | 'TRANSFORMER'       // Transformer
  | 'VAE'               // 变分自编码器
  | 'GAIN';             // 生成对抗插补网络

export type ImputationCategory = 'basic' | 'statistical' | 'timeseries' | 'ml' | 'dl';

export type DatasetType = 'FLUX' | 'WEATHER' | 'SOIL' | 'WATER' | 'GENERAL';
```

## 4. 类型定义

### 4.1 核心类型

```typescript
// 插补结果状态
export type ImputationResultStatus = 
  | 'PENDING'    // 待执行
  | 'RUNNING'    // 执行中
  | 'COMPLETED'  // 已完成
  | 'FAILED'     // 失败
  | 'APPLIED'    // 已应用到数据集
  | 'REVERTED';  // 已撤销

// 插补结果
export interface ImputationResult {
  id: number;
  datasetId: number;
  versionId: number;
  methodId: ImputationMethodId;
  imputationParams: string;    // JSON
  targetColumns: string | null; // JSON数组，null表示全部列
  totalRows: number;
  imputedCount: number;
  imputationRate: number;
  status: ImputationResultStatus;
  progress: number;
  progressMessage?: string;
  resultFilePath?: string;
  executedAt: string;
  completedAt?: string;
  createdAt: string;
  updatedAt: string;
}

// 插补详情
export interface ImputationDetail {
  id: number;
  resultId: number;
  columnName: string;
  rowIndex: number;
  timePoint?: string;
  originalValue: number | null;
  imputedValue: number;
  confidence?: number;
}

// 插补列统计
export interface ImputationColumnStat {
  id: number;
  resultId: number;
  columnName: string;
  missingCount: number;
  imputedCount: number;
  meanBefore: number;
  meanAfter: number;
  stdBefore: number;
  stdAfter: number;
  rmse?: number;
  mae?: number;
}

// 插补进度事件
export interface ImputationProgressEvent {
  resultId: number;
  progress: number;        // 0-100
  stage: 'preparing' | 'training' | 'imputing' | 'validating' | 'saving';
  message: string;
  currentColumn?: string;
  estimatedRemaining?: number; // 预估剩余时间（秒）
}

// 插补请求
export interface ImputationRequest {
  datasetId: string;
  versionId: string;
  methodId: ImputationMethodId;
  params: Record<string, any>;
  targetColumns: string[] | null; // null表示全部列
  validateSplit?: number;         // 验证集比例（0-0.3）
}

// 插补对比数据
export interface ImputationComparison {
  columnName: string;
  timePoints: string[];
  originalValues: (number | null)[];
  imputedValues: number[];
  confidences?: number[];
  statistics: {
    before: { mean: number; std: number; min: number; max: number };
    after: { mean: number; std: number; min: number; max: number };
    rmse?: number;
    mae?: number;
  };
}
```

### 5.1 路由定义

```typescript
// shared/constants/apiRoutes.ts 中添加
export const IMPUTATION_ROUTES = {
  // 方法相关

  EXECUTE: 'imputation/execute',

  CANCEL: 'imputation/cancel',
  
  // 结果相关
  GET_RESULTS: 'imputation/get-results',
  GET_RESULT_DETAIL: 'imputation/get-result-detail',
  GET_COLUMN_STATS: 'imputation/get-column-stats',
  DELETE_RESULT: 'imputation/delete-result',
  
  // 应用/撤销
  APPLY_RESULT: 'imputation/apply-result',
  REVERT_RESULT: 'imputation/revert-result',
  
  // 对比预览
  GET_COMPARISON: 'imputation/get-comparison',
  GET_PREVIEW: 'imputation/get-preview',
  
  // 模型相关（ML/DL）
  TRAIN_MODEL: 'imputation/train-model',
  GET_MODELS: 'imputation/get-models',
  DELETE_MODEL: 'imputation/delete-model',
} as const;
```

### 5.2 API 接口

| 路由 | 方法 | 参数 | 返回值 | 说明 |
|------|------|------|--------|------|
| `get-methods` | GET | `datasetType?: string` | `ImputationMethod[]` | 获取可用方法列表 |
| `get-recommended-methods` | GET | `datasetId, datasetType` | `ImputationMethod[]` | 获取推荐方法 |
| `execute` | POST | `ImputationRequest` | `{ resultId: number }` | 执行插补 |
| `get-progress` | GET | `resultId` | `ImputationProgressEvent` | 获取进度 |
| `cancel` | POST | `resultId` | `boolean` | 取消执行 |
| `get-results` | GET | `datasetId` | `ImputationResult[]` | 获取历史结果 |
| `get-result-detail` | GET | `resultId, columnName?, limit?, offset?` | `ImputationDetail[]` | 获取详情 |
| `get-column-stats` | GET | `resultId` | `ImputationColumnStat[]` | 获取列统计 |
| `delete-result` | DELETE | `resultId` | `boolean` | 删除结果 |
| `apply-result` | POST | `resultId` | `{ versionId: number }` | 应用到数据集 |
| `revert-result` | POST | `resultId` | `boolean` | 撤销应用 |
| `get-comparison` | GET | `resultId, columnName` | `ImputationComparison` | 获取对比数据 |
| `train-model` | POST | `datasetId, modelType, params` | `{ modelId: number }` | 训练模型 |

#### 7.2.2 ImputationService

```typescript
// electron/service/ImputationService.ts
export class ImputationService {
  private repository: ImputationRepository;
  private pythonBridge: PythonBridge;
  private progressMap: Map<number, ImputationProgressEvent>;
  private abortControllers: Map<number, AbortController>;
  
  constructor() {
    this.repository = new ImputationRepository();
    this.pythonBridge = new PythonBridge();
    this.progressMap = new Map();
    this.abortControllers = new Map();
  }
  
  async executeImputation(request: ImputationRequest): Promise<number> {
    // 1. 创建结果记录
    const resultId = await this.repository.createResult({
      datasetId: parseInt(request.datasetId),
      versionId: parseInt(request.versionId),
      methodId: request.methodId,
      imputationParams: JSON.stringify(request.params),
      targetColumns: request.targetColumns ? JSON.stringify(request.targetColumns) : null,
      status: 'RUNNING',
      progress: 0,
    });
    
    // 2. 创建中止控制器
    const abortController = new AbortController();
    this.abortControllers.set(resultId, abortController);
    
    // 3. 异步执行插补
    this.runImputationAsync(resultId, request, abortController.signal);
    
    return resultId;
  }
  
  private async runImputationAsync(
    resultId: number, 
    request: ImputationRequest,
    signal: AbortSignal
  ): Promise<void> {
    try {
      // 更新进度
      this.updateProgress(resultId, 5, 'preparing', '准备数据...');
      
      // 加载数据
      const data = await this.loadData(request.datasetId, request.versionId);
      const columns = request.targetColumns || this.getColumnsWithMissing(data);
      
      // 根据方法类型选择执行策略
      const method = this.getMethod(request.methodId);
      
      if (method.category === 'basic' || method.category === 'statistical') {
        await this.executeBasicImputation(resultId, data, columns, request, signal);
      } else if (method.category === 'timeseries') {
        await this.executeTimeSeriesImputation(resultId, data, columns, request, signal);
      } else if (method.category === 'ml' || method.category === 'dl') {
        await this.executeMLImputation(resultId, data, columns, request, signal);
      }
      
      // 完成
      await this.repository.updateResult(resultId, {
        status: 'COMPLETED',
        progress: 100,
        completedAt: new Date().toISOString(),
      });
      
    } catch (error) {
      if (error.name === 'AbortError') {
        await this.repository.updateResult(resultId, { status: 'FAILED', progressMessage: '已取消' });
      } else {
        await this.repository.updateResult(resultId, { status: 'FAILED', progressMessage: error.message });
      }
    } finally {
      this.abortControllers.delete(resultId);
    }
  }
  
  private async executeMLImputation(
    resultId: number,
    data: any,
    columns: string[],
    request: ImputationRequest,
    signal: AbortSignal
  ): Promise<void> {
    // 调用 Python 脚本
    this.updateProgress(resultId, 10, 'training', '正在训练模型...');
    
    const result = await this.pythonBridge.execute('imputation', {
      method: request.methodId,
      data: data,
      columns: columns,
      params: request.params,
    }, (progress) => {
      this.updateProgress(resultId, progress.percent, progress.stage, progress.message);
    }, signal);
    
    // 保存结果
    await this.saveImputationResults(resultId, result);
  }
  
  async cancelExecution(resultId: number): Promise<boolean> {
    const controller = this.abortControllers.get(resultId);
    if (controller) {
      controller.abort();
      return true;
    }
    return false;
  }
  
  getProgress(resultId: number): ImputationProgressEvent | null {
    return this.progressMap.get(resultId) || null;
  }
  
  private updateProgress(
    resultId: number, 
    progress: number, 
    stage: string, 
    message: string,
    currentColumn?: string
  ): void {
    const event: ImputationProgressEvent = {
      resultId,
      progress,
      stage: stage as any,
      message,
      currentColumn,
    };
    this.progressMap.set(resultId, event);
    
    // 通过 IPC 发送到前端
    this.emitProgress(event);
  }
  
  // ... 其他方法
}
```

## 8. 状态管理 (Pinia Store)

```typescript
// frontend/src/stores/useImputationStore.ts
export const useImputationStore = defineStore('imputation', () => {
  // 状态
  const methods = ref<ImputationMethod[]>([]);
  const recommendedMethods = ref<ImputationMethod[]>([]);
  const results = ref<ImputationResult[]>([]);
  const currentResult = ref<ImputationResult | null>(null);
  const columnStats = ref<ImputationColumnStat[]>([]);
  const comparison = ref<ImputationComparison | null>(null);
  
  const loading = ref(false);
  const executing = ref(false);
  const progress = ref<ImputationProgressEvent | null>(null);
  
  // Actions
  async function loadMethods(datasetType?: string) {
    loading.value = true;
    try {
      const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.GET_METHODS, { datasetType });
      if (res.success) {
        methods.value = res.data;
      }
    } finally {
      loading.value = false;
    }
  }
  
  async function loadRecommendedMethods(datasetId: string, datasetType: string) {
    const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.GET_RECOMMENDED_METHODS, {
      datasetId,
      datasetType,
    });
    if (res.success) {
      recommendedMethods.value = res.data;
    }
  }
  
  async function executeImputation(request: ImputationRequest): Promise<number | null> {
    executing.value = true;
    progress.value = null;
    
    try {
      const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.EXECUTE, request);
      if (res.success) {
        // 开始轮询进度
        startProgressPolling(res.data.resultId);
        return res.data.resultId;
      }
      return null;
    } catch (error) {
      executing.value = false;
      throw error;
    }
  }
  
  function startProgressPolling(resultId: number) {
    const pollInterval = setInterval(async () => {
      const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.GET_PROGRESS, { resultId });
      if (res.success && res.data) {
        progress.value = res.data;
        
        if (res.data.progress >= 100 || ['COMPLETED', 'FAILED'].includes(res.data.stage)) {
          clearInterval(pollInterval);
          executing.value = false;
          await loadResults(currentDatasetId.value);
        }
      }
    }, 1000);
  }
  
  async function loadResults(datasetId: string) {
    loading.value = true;
    try {
      const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.GET_RESULTS, { datasetId });
      if (res.success) {
        results.value = res.data;
      }
    } finally {
      loading.value = false;
    }
  }
  
  async function loadComparison(resultId: number, columnName: string) {
    const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.GET_COMPARISON, {
      resultId,
      columnName,
    });
    if (res.success) {
      comparison.value = res.data;
    }
  }
  
  async function applyResult(resultId: number): Promise<boolean> {
    const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.APPLY_RESULT, { resultId });
    if (res.success) {
      await loadResults(currentDatasetId.value);
      return true;
    }
    return false;
  }
  
  async function revertResult(resultId: number): Promise<boolean> {
    const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.REVERT_RESULT, { resultId });
    if (res.success) {
      await loadResults(currentDatasetId.value);
      return true;
    }
    return false;
  }
  
  async function cancelExecution(resultId: number): Promise<boolean> {
    const res = await window.electronAPI.invoke(IMPUTATION_ROUTES.CANCEL, { resultId });
    if (res.success) {
      executing.value = false;
      return true;
    }
    return false;
  }
  
  function reset() {
    results.value = [];
    currentResult.value = null;
    columnStats.value = [];
    comparison.value = null;
    progress.value = null;
    executing.value = false;
  }
  
  return {
    // State
    methods,
    recommendedMethods,
    results,
    currentResult,
    columnStats,
    comparison,
    loading,
    executing,
    progress,
    
    // Actions
    loadMethods,
    loadRecommendedMethods,
    executeImputation,
    loadResults,
    loadComparison,
    applyResult,
    revertResult,
    cancelExecution,
    reset,
  };
});
```
